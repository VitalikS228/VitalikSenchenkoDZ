void task2() {
	unsigned i, n;
	
	printf("n= ");
	scanf_s("%u", &n);

	printf("%u! = ", n);
	for (i = n; i > 0; i--) {
		if (i == 1) {
			printf("%d", i);
			break;
		}
		printf("%d*", i);
	}
	printf("\n");

	printf("%u! = ", n);
	for (i = 1; i <= n; i++) {
		if (i == n) {
			printf("%d", i);
			break;
		}
		printf("%d*", i);
	}
	printf("\n");
}

void task3a() {
	unsigned i, n;
	float x, y, power;

	printf("n=");
	scanf_s("%u", &n);

	printf("x=");
	scanf_s("%f", &x);

	y = 1;
	power = 1;

	for (i = 0; i < n; i++) {
		power *= x;
		printf("%lf\n", power);
		y += power;
	}

	printf("y=%f", y);
}

void task3b() {
	unsigned n = 4, x = 1, y = 2;
	double sum = 0;
	for (int i = n; i >= 0; i--) {
		sum += pow(x, 2 * pow(2, i)) * pow(y, i);

}

void task4(int n, double x) {

}

void task5(int n) {

		}
	}
	else {

}
